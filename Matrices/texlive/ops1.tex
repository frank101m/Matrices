\large{Dada la matriz $A$}
\[
A
=
\opmatrixA
\]
\begin{enumerate}
\item \large{\textbf{Transpuesta de $\mathbf{A}$}}
\[
A^T
=
\opmatrixAT
\]

\item \large{\textbf{Determinante de $\mathbf{A}$}}
\[
|A|=
\opmatrixdetA
=
\opvaldetA
\]
\item \large{\textbf{Inversa de $\mathbf{A}$}}
\[
A^{-1}=
\opmatrixinvA
\]
\end{enumerate}
